def colorize(material)
  @saved_material = faces.map(&:material)
  faces.each { |f| f.material = material}
end

def distance
  return unless a && b
  distance = a.points.first.distance_to_plane(b.plane)
  distance.round(Vectorize::PRECISION)
end
